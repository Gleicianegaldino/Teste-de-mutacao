def is_leap(year):
    '''
        Checks whether a given year is leap year
        @param year: the year.
        @return: True if the year is leap and False otherwise.
    '''
    
    if year <= 1752:
        
        if year % 4 == 0:
            return True
        else:
            return False
    else:
        
        
        if year % 400 == 0:
            return True
        if year % 100 == 0:
            return False
        if year % 4 == 0:
            return True
    
    return False
